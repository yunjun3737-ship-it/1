loadstring(game:HttpGet("https://raw.githubusercontent.com/BlizTBr/scripts/main/FTAP.lua"))()task.spawn(function()
    while task.wait(3) do
        pcall(function()
            for _,v in pairs(game.CoreGui:GetDescendants()) do
                if v:IsA("TextLabel") then
                    local t = tostring(v.Text)

                    if string.find(t,"Bliz") or string.find(t,"HUB") then
                        v.Text = "파카허브(존나힘드네 미공개용)"
                    end
                end
            end
        end)
    end
end)
